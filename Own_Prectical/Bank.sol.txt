contract Bank{
    int bal;
    constructor() public {
        bal=1;
    }
    function getBalance() view public returns(int){
        return bal;
    }
    function Withdorow(int amount) public{
        bal=bal-amount;
    }
    function deposit(int amt) public{
        bal=bal+amt;

}
